4. create a basic class for finding records

<?php

// uses MySQLi but needs to be written for PDO
class Model {

public static function find(integer $id, $select = '*') {

$class = get_called_class();
$table = $class::$table_name;

$dbh = getdbh();

$query = "SELECT $select FROM $table WHERE `id` = ?";

if( $stmt = $dbh->prepare($query) ) {

$stmt->bindParam('i', $id);

$stmt->execute();

return $stmt->fetch();
}

return false;
} // end static::find()

}// end Model
?>
